sudo [command] - выполнить команду от имени админа
mkdir [dir_name] - создание директории
rm [file] - удаление файла по его директории относительно текущей
ls - список директорий и файлов в текущей директории
cd [dir]- сменить директорию
	(cd ..) - вверх по дереву каталогов
	(cd - ) - вернуться в предыдущую директорию
pwd - возвращает путь текущей директории
nano [file] - редактирование текстового файла с помощью nano
vim [file] - редактирование текстового файла с помощью vim
touch [file] - создание файла
>[file] - создание файла в текущей директории
cp [file] [dir] - 
mv [file] [dir] - 
zip [name.zip] [files] - сжимает files
unzip [file.zip] -d [path_extract] - распаковывает содержимое file в указанную директорию
apt [options]- команда утилиты управления пакетов
	options - remove, install upgrade, update
hostname - используется для просмотра имени компьютера и имени домена
